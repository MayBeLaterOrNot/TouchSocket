## 一、调用上下文

在 WebAPI 方法中，可以通过 `IWebApiCallContext` 访问请求的详细信息。

<BilibiliCard title="获取调用上下文" link="https://www.bilibili.com/cheese/play/ep1688332" isPro="true"/>

### 1.1 获取调用上下文

```csharp showLineNumbers
public partial class ApiServer : SingletonRpcServer
{
    [WebApi(Method = HttpMethodType.Get)]
    public string GetClientInfo(IWebApiCallContext callContext)
    {
        var ip = callContext.Caller.IP;
        var port = callContext.Caller.Port;
        return $"Client: {ip}:{port}";
    }
}
```

### 1.2 访问 HTTP 上下文

通过调用上下文访问完整的 HTTP 请求和响应：

```csharp showLineNumbers
[WebApi(Method = HttpMethodType.Get)]
public string GetRequestInfo(IWebApiCallContext callContext)
{
    var httpContext = callContext.HttpContext;
    var request = httpContext.Request;
    var response = httpContext.Response;

    // 获取请求信息
    var method = request.Method;
    var url = request.URL;
    var userAgent = request.Headers["User-Agent"];

    // 设置响应头
    response.SetHeader("X-Custom-Header", "CustomValue");

    return $"Method: {method}, URL: {url}";
}
```

public async Task<string> UploadFile(IWebApiCallContext callContext)

    {
        // 保存文件

### 5.2 多文件上传

```csharp showLineNumbers
[WebApi(Method = HttpMethodType.Post)]
public async Task<string> UploadMultipleFiles(IWebApiCallContext callContext)
{
    var request = callContext.HttpContext.Request;
    
    if (request.TryGetContent(out var content))
    {
        using var multifileCollection = new MultifileCollection(content);
        
        var count = 0;
        foreach (var multifile in multifileCollection)
        {
            var fileName = multifile.FileName;
            var filePath = Path.Combine("uploads", fileName);
            
            Directory.CreateDirectory("uploads");
            using var fs = File.Create(filePath);
            await multifile.DataSection.WriteTo Async(fs);
            
            count++;
        }
        
        return $"Uploaded {count} files";
    }
    
    return "No files";
}
```

### 5.3 文件下载

<BilibiliCard title="文件下载" link="https://www.bilibili.com/cheese/play/ep1688334" isPro="true"/>

```csharp showLineNumbers
[WebApi(Method = HttpMethodType.Get)]
public async Task DownloadFile(IWebApiCallContext callContext)
{
    var response = callContext.HttpContext.Response;
    var filePath = "files/document.pdf";
    
    if (File.Exists(filePath))
    {
        response.SetStatus(200, "OK");
        response.ContentType = "application/octet-stream";
        response.SetHeader("Content-Disposition", "attachment; filename=document.pdf");
        
        var fileData = await File.ReadAllBytesAsync(filePath);
        response.SetContent(fileData);
        
        await response.AnswerAsync();
    }
    else
    {
        response.SetStatus(404, "Not Found");
        await response.AnswerAsync();
    }
}
```

## 六、WebSocket 升级

可以将 HTTP 连接升级为 WebSocket：

```csharp showLineNumbers
[WebApi(Method = HttpMethodType.Get)]
public async Task UpgradeToWebSocket(IWebApiCallContext callContext)
{
    var httpContext = callContext.HttpContext;
    
    if (httpContext.Request.IsUpgrade())
    {
        if (await httpContext.Response.SwitchProtocolsAsync())
        {
            var webSocket = httpContext.WebSocket;
            
            // 接收 WebSocket 消息
            using var receiveBlock = new ValueByteBlock();
            while (true)
            {
                var result = await webSocket.ReadAsync(receiveBlock);
                if (result.IsCompleted)
                {
                    break;
                }
                
                // 回显消息
                await webSocket.SendAsync(receiveBlock.Memory);
                receiveBlock.Reset();
            }
        }
    }
}
```

## 七、完整示例

```csharp showLineNumbers
public partial class FileApiServer : SingletonRpcServer
{
    // 获取客户端信息
    [WebApi(Method = HttpMethodType.Get)]
    public string GetInfo(IWebApiCallContext callContext)
    {
        var caller = callContext.Caller;
        var request = callContext.HttpContext.Request;
        
        return new 
        {
            ClientIp = caller.IP,
            ClientPort = caller.Port,
            Method = request.Method,
            Path = request.RelativeURL,
            UserAgent = request.Headers["User-Agent"]
        }.ToJsonString();
    }

    // 文件上传
    [WebApi(Method = HttpMethodType.Post)]
    public async Task<string> Upload(IWebApiCallContext callContext)
    {
        var request = callContext.HttpContext.Request;
        var fileName = request.Query.Get("filename") ?? "upload.dat";
        
        if (request.TryGetContent(out var content))
        {
            var filePath = Path.Combine("uploads", fileName);
            Directory.CreateDirectory("uploads");
            
            await File.WriteAllBytesAsync(filePath, content);
            return $"Uploaded: {fileName} ({content.Length} bytes)";
        }
        
        return "Upload failed";
    }

    // 文件下载
    [WebApi(Method = HttpMethodType.Get)]
    public async Task Download(IWebApiCallContext callContext, string filename)
    {
        var response = callContext.HttpContext.Response;
        var filePath = Path.Combine("files", filename);
        
        if (File.Exists(filePath))
        {
            response.SetStatus(200, "OK");
            response.ContentType = "application/octet-stream";
            response.SetHeader("Content-Disposition", $"attachment; filename={filename}");
            
            var fileData = await File.ReadAllBytesAsync(filePath);
            response.SetContent(fileData);
        }
        else
        {
            response.SetStatus(404, "Not Found");
            response.SetContent("File not found");
        }
        
        await response.AnswerAsync();
    }
}
```
